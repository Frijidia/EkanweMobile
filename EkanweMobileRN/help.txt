démarrer l'émulateur Android

    emulator -list-avds
    emulator -avd [nom_de_votre_avd]

lancer l'application React Native 

    cd EkanweMobileRN
    npx react-native run-android

nettoyer le cache

    cd android
    ./gradlew clean
    cd ..
    npx react-native run-android

vérifier que Metro bundler est bien lancé

    npx react-native start

mkdir ../backup
cp -r src ../backup/
cp package.json ../backup/

cd ..
rm -rf EkanweMobileRN
npx react-native init EkanweMobileRN

cp -r backup/src EkanweMobileRN/
cp backup/package.json EkanweMobileRN/
cd EkanweMobileRN
npm install
